function renderCards(items) { const container = document.getElementById('dashboard'); container.innerHTML = ''; items.forEach(item => { const card = document.createElement('div'); card.className = 'link-card'; card.dataset.tags = JSON.stringify(item.tags || []); const tagHtml = (item.tags || []).map(t => `${t}`).join(''); // 處理地址顯示邏輯：如果 note 存在，則顯示並製作地圖連結 let addressHtml = ''; if (item.note && item.note.trim() !== "") { const encodedAddress = encodeURIComponent(item.note); const mapUrl = `https://www.google.com/maps/search/?api=1&query=${encodedAddress}`; addressHtml = `
📍 地址： ${item.note}
`; } card.innerHTML = `
${item.title}
${tagHtml}
${addressHtml} 立即前往
`; container.appendChild(card); }); }
