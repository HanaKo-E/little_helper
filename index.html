function renderFilters(items) { const filterBar = document.getElementById('filter-bar'); const primaryTags = new Set(); items.forEach(item => { if (item.tags && item.tags.length > 0) primaryTags.add(item.tags[0]); }); const tagArray = Array.from(primaryTags); // 清空除了「全部」以外的按鈕，避免重複渲染 while (filterBar.childNodes.length > 2) { filterBar.removeChild(filterBar.lastChild); } tagArray.forEach((tag, index) => { const btn = document.createElement('button'); btn.className = 'filter-btn'; btn.innerText = tag; btn.onclick = () => filterContent(tag, btn); filterBar.appendChild(btn); }); // 初始化所有按鈕顏色 updateFilterStyles(); } // 新增：統一管理按鈕樣式的函數 function updateFilterStyles() { const btns = document.querySelectorAll('.filter-btn'); const total = btns.length; btns.forEach((btn, index) => { if (btn.classList.contains('active')) { // 選中狀態：固定白底深字 btn.style.backgroundColor = 'white'; btn.style.color = '#004d40'; btn.style.borderColor = '#004d40'; } else { // 非選中狀態：根據索引計算漸層透明度 const opacity = Math.max(0.3, 0.95 - (index / total) * 0.7); btn.style.backgroundColor = `rgba(0, 77, 64, ${opacity})`; btn.style.color = opacity < 0.5 ? '#004d40' : 'white'; btn.style.borderColor = 'transparent'; } }); } function filterContent(tag, btnEl) { // 1. 切換 Active 類名 document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active')); btnEl.classList.add('active'); // 2. 重新整理所有按鈕的視覺樣式 updateFilterStyles(); // 3. 過濾卡片內容 document.querySelectorAll('.link-card').forEach(card => { card.style.display = (tag === 'all' || card.dataset.primaryTag === tag) ? "" : "none"; }); }
