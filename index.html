/* --- 請替換原本的 renderFilters 函數 --- */ function renderFilters(items) { const filterBar = document.getElementById('filter-bar'); const primaryTags = new Set(); items.forEach(item => { if (item.tags && item.tags.length > 0) primaryTags.add(item.tags[0]); }); const tagArray = Array.from(primaryTags); const total = tagArray.length + 1; // 加上「全部」按鈕 // 處理「全部」按鈕 (最深) const btnAll = document.getElementById('btn-all'); btnAll.style.backgroundColor = 'rgba(0, 77, 64, 1)'; btnAll.style.color = 'white'; btnAll.onclick = () => filterContent('all', btnAll); tagArray.forEach((tag, index) => { const btn = document.createElement('button'); btn.className = 'filter-btn'; btn.innerText = tag; // 動態計算透明度：從 0.95 逐漸降到 0.3 // 公式：1 - (當前索引 / 總數) * 比例 const opacity = Math.max(0.3, 0.95 - ((index + 1) / total) * 0.8); btn.style.backgroundColor = `rgba(0, 77, 64, ${opacity})`; btn.style.color = opacity < 0.5 ? '#004d40' : 'white'; // 太淺時字改用深色確保讀取 btn.style.borderColor = 'transparent'; btn.onclick = () => filterContent(tag, btn); filterBar.appendChild(btn); }); } /* --- 同步更新 filterContent 確保點擊後的狀態正確 --- */ function filterContent(tag, btnEl) { document.querySelectorAll('.filter-btn').forEach(b => { b.classList.remove('active'); // 恢復原本該有的透明度（這裡我們從 style 屬性重新讀取或設回初始） // 簡單做法：讓 active 狀態直接反轉顏色 }); btnEl.classList.add('active'); // active 狀態強制變成白底深字，確保選中感 btnEl.style.backgroundColor = 'white'; btnEl.style.color = '#004d40'; btnEl.style.borderColor = '#004d40'; document.querySelectorAll('.link-card').forEach(card => { card.style.display = (tag === 'all' || card.dataset.primaryTag === tag) ? "" : "none"; }); }
